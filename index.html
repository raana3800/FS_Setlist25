function renderSongs(songList) { const tableBody = document.getElementById('tableBody'); tableBody.innerHTML = ''; songList.forEach((song, index) => { const row = document.createElement('tr'); row.innerHTML = ` ≡ ${index + 1} ${song.title} ✗ ✗ ✗ ✗ ✗ `; tableBody.appendChild(row); }); new Sortable(tableBody, { handle: '.drag-handle', animation: 150 }); document.querySelectorAll('.bpm-input').forEach(input => { input.addEventListener('change', () => { input.classList.add('highlight'); setTimeout(() => input.classList.remove('highlight'), 600); }); }); document.querySelectorAll('.status').forEach(status => { status.addEventListener('click', () => { status.classList.add('clicked'); setTimeout(() => status.classList.remove('clicked'), 300); if (status.classList.contains('red')) { status.classList.remove('red'); status.classList.add('green'); status.innerText = '✓'; } else { status.classList.remove('green'); status.classList.add('red'); status.innerText = '✗'; } }); }); const style = document.createElement('style'); style.innerHTML = ` .clicked { animation: pop 0.3s; } @keyframes pop { 0% { transform: scale(1); } 50% { transform: scale(1.4); } 100% { transform: scale(1); } } `; document.head.appendChild(style); }
